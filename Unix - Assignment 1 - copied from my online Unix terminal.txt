~$ cat << 'EOF' > my_parking_script.sh
#!/bin/bash

# Check if the csv file is provided
if [ -z "$1" ]; then
  echo "Please provide the path to the csv file as a positional parameter."
  exit 1
fi

# Function to print all types of infraction_description
print_infraction_descriptions() {
  awk -F',' 'NR>1{a[$4]}END{for(i in a) print i}' "$1"
}

# Function to print the mean, min, and max set_fine_amount
print_set_fine_stats() {
  awk -F',' 'NR>1{sum+=$5; a[NR-1]=$5}END{count=NR-2; mean=sum/count;min=max=a[2]; for(i=2;i<=count;i++){if(a[i]<min) min=a[i]; if(a[i]>max) max=a[i]}; print "Mean: " mean, "Min: " min, "Max: " max}' "$1"
}

# Function to copy selected fields to the new CSV file
copy_fields_to_csv() {
  awk -F',' 'NR>1 && $4 == "PARK ON PRIVATE PROPERTY" {print $4,$5,$8}' "$1" > Leila_chosen_infraction.csv
}

# Call the function to print all types of infraction_description
print_infraction_descriptions "$1"

# Call the function to print the mean, min, and max set_fine_amount
print_set_fine_stats "$1"

# Call the function to copy selected fields to the new CSV file
copy_fields_to_csv "$1"
EOF
